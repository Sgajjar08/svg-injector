import cloneSvg from './clone-svg'
import svgCache from './svg-cache'
 
type Callback = (error: Error | null, svg?: SVGSVGElement) => undefined
type RequestQueue = Map<string, Callback[]>
 
const requestQueue: RequestQueue = new Map()
 
export const queueRequest = (url: string, callback: Callback) => {
  requestQueue.set(url, [...(requestQueue.get(url) || []), callback])
}
 
export const processRequestQueue = (url: string) => {
  const callbackArray = requestQueue.get(url)
 
  if (!callbackArray) {
    return
  }
 
  for (let i = 0, j = callbackArray.length; i < j; i++) {
    setTimeout(() => {
      const callback = callbackArray[i]
      const cacheValue = svgCache.get(url)
 
      if (cacheValue instanceof SVGSVGElement) {
        callback(null, cloneSvg(cacheValue))
        return
      }
 
      if (cacheValue instanceof Error) {
        callback(cacheValue)
        return
      }
    }, 0)
  }
}
